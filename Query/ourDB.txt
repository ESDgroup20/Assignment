CLIENTID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (start with 1, increment by 1), 
CLIENTNAME VARCHAR(50), 
CLIENTADDRESS VARCHAR(100), 
CLIENTTYPE VARCHAR(10), 
UNAME VARCHAR(20),
